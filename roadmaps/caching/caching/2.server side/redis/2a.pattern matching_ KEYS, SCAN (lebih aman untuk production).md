async function scanAllKeys(pattern) {
  let cursor = "0";
  let keys = [];

  do {
    const result = await redis.scan(cursor, "MATCH", pattern, "COUNT", 100);
    cursor = result[0];

  } while (cursor !== "0");

  return keys;
}
